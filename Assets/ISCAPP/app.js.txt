// --- STATE VARIABLES ---
let questions = [];
let currentQuestionIndex = 0;
let score = 0;
let streak = 0;
let rank = "Intern";

// Points awarded per correct answer
const POINTS_PER_QUESTION = 10; 

// --- DOM ELEMENTS ---
const elements = {
    rankDisplay: document.getElementById('user-rank'),
    scoreDisplay: document.getElementById('score-display'),
    streakDisplay: document.getElementById('streak-display'),
    progressBar: document.getElementById('progress-bar'),
    currentQNum: document.getElementById('current-q-num'),
    totalQNum: document.getElementById('total-q-num'),
    areaTag: document.getElementById('content-area-tag'),
    skillTag: document.getElementById('skill-level-tag'),
    topicTag: document.getElementById('topic-tag'),
    questionText: document.getElementById('question-text'),
    optionsContainer: document.getElementById('options-container'),
    explanationBox: document.getElementById('explanation-box'),
    explanationTitle: document.getElementById('explanation-title'),
    explanationText: document.getElementById('explanation-text'),
    referenceText: document.getElementById('reference-text'),
    nextBtn: document.getElementById('next-btn')
};

// --- INITIALIZATION ---
async function initApp() {
    loadUserData();
    await fetchQuestions();
    
    if (questions.length > 0) {
        elements.totalQNum.innerText = questions.length;
        loadQuestion();
    } else {
        elements.questionText.innerText = "Error: Could not load questions.";
    }
}

// --- DATA FETCHING ---
async function fetchQuestions() {
    try {
        const response = await fetch('questions.json');
        const data = await response.json();
        // We only need the questions array from your JSON structure
        questions = data.questions; 
    } catch (error) {
        console.error("Failed to load questions.json:", error);
    }
}

// --- GAMIFICATION & LOCAL STORAGE ---
function loadUserData() {
    // Retrieve data from localStorage (or default to 0 if it's their first time)
    score = parseInt(localStorage.getItem('isc_score')) || 0;
    streak = parseInt(localStorage.getItem('isc_streak')) || 0;
    
    // Streak Logic: Check if they logged in yesterday
    const lastLogin = localStorage.getItem('isc_last_login');
    const today = new Date().toDateString();
    
    if (lastLogin !== today) {
        const yesterday = new Date();
        yesterday.setDate(yesterday.getDate() - 1);
        
        if (lastLogin === yesterday.toDateString()) {
            streak++; // Logged in yesterday, increase streak!
            fireConfetti();
        } else if (lastLogin) {
            streak = 0; // Missed a day, reset streak.
        }
        localStorage.setItem('isc_last_login', today);
        localStorage.setItem('isc_streak', streak);
    }

    updateRank();
    updateDashboard();
}

function saveScore() {
    localStorage.setItem('isc_score', score);
}

function updateRank() {
    const oldRank = rank;
    
    // CPA-themed ranking tiers
    if (score >= 300) rank = "IT Audit Partner";
    else if (score >= 150) rank = "Manager";
    else if (score >= 80) rank = "Senior Associate";
    else if (score >= 30) rank = "Staff Auditor";
    else rank = "Intern";

    // Trigger big confetti if they ranked up!
    if (oldRank !== rank && score > 0) {
        fireConfetti(true); 
    }
}

function updateDashboard() {
    elements.scoreDisplay.innerText = score;
    elements.streakDisplay.innerText = `${streak} 🔥`;
    elements.rankDisplay.innerText = rank;
}

// --- CORE QUIZ LOGIC ---
function loadQuestion() {
    // Reset UI for new question
    elements.explanationBox.classList.add('hidden');
    elements.optionsContainer.innerHTML = '';
    
    const currentQ = questions[currentQuestionIndex];
    
    // Update Progress
    elements.currentQNum.innerText = currentQuestionIndex + 1;
    const progressPercent = ((currentQuestionIndex) / questions.length) * 100;
    elements.progressBar.style.width = `${progressPercent}%`;

    // Populate Metadata Badges
    elements.areaTag.innerText = currentQ.content_area;
    elements.skillTag.innerText = currentQ.skill_level;
    elements.topicTag.innerText = currentQ.topic;

    // Set Question Text
    elements.questionText.innerText = currentQ.question_text;

    // Create Option Buttons dynamically
    currentQ.options.forEach((optionText, index) => {
        const button = document.createElement('button');
        button.classList.add('option-btn');
        button.innerText = optionText;
        button.onclick = () => selectOption(index, button);
        elements.optionsContainer.appendChild(button);
    });
}

function selectOption(selectedIndex, selectedButton) {
    const currentQ = questions[currentQuestionIndex];
    const isCorrect = selectedIndex === currentQ.correct_answer_index;
    const allButtons = document.querySelectorAll('.option-btn');

    // Disable all buttons so the user can't click again
    allButtons.forEach(btn => btn.disabled = true);

    // Apply CSS classes for visual feedback
    if (isCorrect) {
        selectedButton.classList.add('correct');
        score += POINTS_PER_QUESTION;
        saveScore();
        updateRank();
        updateDashboard();
        elements.explanationTitle.innerText = "Correct! ✅";
        elements.explanationTitle.style.color = "var(--success-green)";
    } else {
        selectedButton.classList.add('incorrect');
        // Highlight the actual correct answer
        allButtons[currentQ.correct_answer_index].classList.add('correct');
        elements.explanationTitle.innerText = "Incorrect ❌";
        elements.explanationTitle.style.color = "var(--error-red)";
    }

    // Populate and show the explanation box based on the specific option clicked
    elements.explanationText.innerText = currentQ.option_explanations[selectedIndex];
    elements.referenceText.innerText = currentQ.reference_material;
    elements.explanationBox.classList.remove('hidden');
}

// --- NAVIGATION ---
elements.nextBtn.addEventListener('click', () => {
    currentQuestionIndex++;
    
    if (currentQuestionIndex < questions.length) {
        loadQuestion();
    } else {
        // End of the quiz
        elements.progressBar.style.width = '100%';
        elements.questionText.innerText = "Quiz Complete! Great job studying for the ISC Exam.";
        elements.optionsContainer.innerHTML = '';
        elements.explanationBox.classList.add('hidden');
        elements.questionMetadata = document.querySelector('.question-metadata').classList.add('hidden');
        fireConfetti(true);
    }
});

// --- EFFECTS ---
function fireConfetti(isBigBlast = false) {
    if (typeof confetti === 'function') {
        if (isBigBlast) {
            // Huge blast for leveling up or finishing
            confetti({ particleCount: 150, spread: 100, origin: { y: 0.6 } });
        } else {
            // Smaller blast for daily streaks
            confetti({ particleCount: 50, spread: 60, origin: { y: 0.6 } });
        }
    }
}

// Start the app!
initApp();
